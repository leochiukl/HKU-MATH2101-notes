\subsection*{\Cref{sect:vector-spaces}}
\item \Cref{prp:vector-space-prop}: properties of a vector space
\item \Cref{prp:vec-subsp-contain-zero}: vector subspace always contains zero vector
\item \Cref{prp:vec-subsp-is-vec-sp}: vector subspace is a vector space
\item \Cref{prp:span-vec-subsp}: span is a vector subspace
\item \labelcref{it:col-space-expr}: expression for column space
\item \labelcref{it:two-nonzero-vec-lin-dep}: equivalent definition of linear dependence for two nonzero vectors
\item \labelcref{it:lin-dep-subsets}: relationship between linear (in)dependence of subsets
\item \labelcref{it:span-set-not-smaller-than-basis}: spanning set cannot be strictly smaller than basis
\item \labelcref{it:span-set-greater-than-basis-ld}: spanning set strictly greater than basis is not linearly independent
\item \Cref{thm:basis-unique-lin-comb}: criterion for being a basis (unique linear combination)
\item \Cref{prp:li-num-leq-dim}: cardinality of linear independent set is less than or equal to the dimension
\item \Cref{cor:two-bases-same-num}: same cardinality for two bases
\item \labelcref{it:dim-span}: dimension of span
\item \labelcref{it:dim-subspace}: dimension of vector subspace
\item \Cref{thm:add-vec-not-in-spn}: adding a vector not in span preserves linear independence
\item \Cref{thm:card-dim-li-basis}: linear independent set having cardinality equal to dimension must be basis
\item \Cref{thm:remove-vec-still-spn}: possibility of removing a vector from a linearly dependent set while preserving the span
\item \Cref{lma:basis-subset}: existence of basis for \(\spn{S}\) in finite set \(S\)
\item \Cref{thm:card-dim-spn-basis}: spanning set having cardinality equal to dimension must be basis
\item \Cref{thm:matx-inv-crit}: criteria for matrix invertibility
\item \Cref{thm:mult-inv-matx-preserv-li}: preservation of linear independence after multiplication of invertible matrix
\item \Cref{thm:mult-inv-matx-preserv-dim}: preservation of dimension of column space after multiplication of invertible matrix
\item \Cref{prp:mult-matx-reduce-col-sp-dim}: reduction of dimension of column space after multiplication of matrix
\item \Cref{thm:find-col-space-basis}: columns corresponding to the leading ones in the RREF form a basis for column space
\item \Cref{cor:dim-col-space-num-lead-ones}: dimension of column space is the number of leading ones in the RREF
\item \Cref{thm:mult-inv-matx-preserv-row-space}: preservation of row space after multiplication of invertible matrix
\item \Cref{prp:mult-matx-reduce-row-sp-dim}: reduction of dimension of row space after multiplication of matrix
\item \Cref{thm:nonzero-rows-basis-row-sp}: nonzero rows of the RREF form a basis for row space
\item \Cref{cor:dim-row-space-num-lead-ones}: dimension of row space is the number of leading ones in the RREF
\item \Cref{thm:row-col-same-dim}: equality of dimensions of column and row spaces
\item \Cref{cor:matx-trans-same-rk}: taking matrix transpose does not change rank
\item \Cref{thm:null-sp-vec-subsp}: null space is a vector subspace
\item \Cref{thm:null-sp-dim-basis}: dimension and basis of null space
